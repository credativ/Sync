% New table styles
\newcolumntype{V}[1]{%                          % Flattersatz für p; Trennung im ersten Wort möglich
        >{\RaggedRight\arraybackslash\hspace{0pt}}p{#1}%
}
\newcolumntype{v}[1]{%                          % Flattersatz für p; keine Trennung
        >{\raggedright \arraybackslash}p{#1}%
}
\newcolumntype{M}[1]{%                          % Flattersatz für m; Trennung im ersten Wort möglich
        >{\RaggedRight\arraybackslash\hspace{0pt}}m{#1}%
}
\newcolumntype{R}[1]{%                          % Spaltenkopf um 90° gedreht
        >{\begin{turn}{90}\begin{minipage}{#1}%
                \RaggedRight\arraybackslash\hspace{0pt}}l
        <{\end{minipage}\end{turn}}%
}
\newcommand*{\armultirow}[3]{%          % Ermöglich das Zusammenfassen von mehreren Zeilen
        \multicolumn{#1}{#2}{%
        \begin{picture}(0,0)%
                \put(0,0){\begin{tabular}[t]{@{}#2@{}}#3\end{tabular}}%
        \end{picture}%
        }%
}
\makeatletter
\newcolumntype{K}[1]{%                          % Zahlen zuerst am Dezimaltrenner, dann linksbündig
        >{\DC@{.}{.}{#1}}l<{\DC@end}%
}
\makeatother
